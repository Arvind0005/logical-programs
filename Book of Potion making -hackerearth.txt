int main(){
	long long int num;
	int mult=10,result=0;
	scanf("%lli", &num);             			// Reading input from STDIN
	while(num>0){
	    result+=(num%10)*mult;
	    num/=10;
	    mult--;
	} 
	if(result%11==0)
	    printf("Legal ISBN");
	else
	    printf("Illegal ISBN");// Writing output to STDOUT
}
